Arrays in Programming: A Fundamental Building Block In the realm of programming, arrays stand out as fundamental structures that play a crucial role in organizing and manipulating data. An array is a collection of elements, each identified by an index or a key. These elements can be of the same data type, forming a homogeneous container, or a mix of different types in a heterogeneous array. Arrays provide a powerful means of storing and accessing data efficiently. Their ability to store multiple values under a single variable name simplifies the code, making it more readable and maintainable. Moreover, arrays facilitate iteration, allowing developers to perform operations on each element systematically.

          void findMostDuplicates(long long num) {
              int digitCount[10] = {0}; 
              int maxDigit = -1;
              int maxCount = 0;

              while (num > 0) {
                  int digit = num % 10;
                  digitCount[digit]++;
                  num /= 10;
              }

              for (int i = 0; i < 10; i++) {
                  if (digitCount[i] > maxCount) {
                      maxCount = digitCount[i];
                      maxDigit = i;
                  }
              }

              printf("Paling banyak duplikasi angka %d karena ada %d\\n", maxDigit, maxCount);
          }

          int main() {
              long long num;

              printf("Masukkan sebuah angka: ");
              scanf("%lld", &num);

              findMostDuplicates(num);

              return 0;
          }

        #include <stdio.h>

        #define MAX_ROWS 100
        #define MAX_COLS 100

        void printMatrix(int matrix[MAX_ROWS][MAX_COLS], int rows, int cols) {
            printf("Modified Matrix:\n");
            for (int i = 0; i < rows; i++) {
                for (int j = 0; j < cols; j++) {
                    printf("%d\t", matrix[i][j]);
                }
                printf("\n");
            }
        }

        void replaceElement(int matrix[MAX_ROWS][MAX_COLS], int rows, int cols, int x, int y, int newValue) {
            if (x >= 0 && x < rows && y >= 0 && y < cols) {
                matrix[x][y] = newValue;
            } else {
                printf("Invalid index.\n");
            }
        }

        int main() {
            int matrix[MAX_ROWS][MAX_COLS];
            int rows, cols, x, y, newValue;

            printf("Enter the number of rows and columns (space-separated): ");
            scanf("%d %d", &rows, &cols);

            printf("Enter the matrix elements:\n");
            for (int i = 0; i < rows; i++) {
                for (int j = 0; j < cols; j++) {
                    scanf("%d", &matrix[i][j]);
                }
            }

            printf("Enter the index (x y) and the new value (space-separated): ");
            scanf("%d %d %d", &x, &y, &newValue);

            replaceElement(matrix, rows, cols, x, y, newValue);

            printMatrix(matrix, rows, cols);

            return 0;
        }
